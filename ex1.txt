# Ex-1: Univariate, Bivariate, Multivariate Regression & Analysis

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error

# 🔹 Load dataset
df = pd.read_csv("data.csv")  # change file name

# 🔹 Drop missing values
df = df.dropna()

# -------- UNIVARIATE REGRESSION --------
X_uni = df.iloc[:, [0]]  # only one feature
y = df.iloc[:, -1]

X_train, X_test, y_train, y_test = train_test_split(X_uni, y, test_size=0.2)

model = LinearRegression()
model.fit(X_train, y_train)
y_pred = model.predict(X_test)

print("Univariate MSE:", mean_squared_error(y_test, y_pred))

# -------- BIVARIATE REGRESSION --------
X_bi = df.iloc[:, [0, 1]]  # two features

X_train, X_test, y_train, y_test = train_test_split(X_bi, y, test_size=0.2)

model = LinearRegression()
model.fit(X_train, y_train)
y_pred = model.predict(X_test)

print("Bivariate MSE:", mean_squared_error(y_test, y_pred))

# -------- MULTIVARIATE REGRESSION --------
X_multi = df.iloc[:, :-1]  # all features except target

X_train, X_test, y_train, y_test = train_test_split(X_multi, y, test_size=0.2)

model = LinearRegression()
model.fit(X_train, y_train)
y_pred = model.predict(X_test)

print("Multivariate MSE:", mean_squared_error(y_test, y_pred))
